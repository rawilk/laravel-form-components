<div x-data="{
        onValue: true,
        offValue: false,
                    value: false,
                get isPressed() {
            return this.value === this.onValue;
        },
        toggle() {
            this.value = this.isPressed ? this.offValue : this.onValue;
        },
     }"
     wire:ignore.self
     class="flex items-center "
     
>
    
    <button x-bind:aria-pressed="JSON.stringify(isPressed)"
            x-on:click="toggle()"
            x-ref="button"
            x-cloak
            type="button"
             id="foo"                         class="switch-toggle switch-toggle--base switch-toggle-simple foo-class" data-foo="bar"
            x-bind:class="{ 'pressed': isPressed }"
                >
        <span class="sr-only">Turn on</span>

                    <span aria-hidden="true"
                  class="switch-toggle-button"
                  x-bind:class="{ 'pressed': isPressed }"
            >
                
                            </span>
            </button>

    
    </div>
